{%- comment -%} Prints read time. Receives an article. {%- endcomment -%} {%- assign current = include.article -%} {%- unless current -%} {%- assign current = page -%} {%- endunless -%} {%- assign words = current.content | number_of_words -%} {%- if words < 300 -%} {{- "1 min read" -}} {%- else -%} {{- words | divided_by:300 | append: " min read" -}} {%- endif -%}
